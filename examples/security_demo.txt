# Security features demonstration

# Hash a password
store → password → "my_secret_password"
store → hash → sha256(password)
print → "Password hash: " + hash

# Encrypt data
store → data → "sensitive information"
store → key → "my_encryption_key"
store → encrypted → encrypt(data, key)
print → "Encrypted: " + encrypted

# Decrypt data
store → decrypted → decrypt(encrypted, key)
print → "Decrypted: " + decrypted

# Generate random data
store → random_data → random_bytes(32)
print → "Random bytes: " + random_data
