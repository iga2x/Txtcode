# Port scanner example in Txt-code
# This demonstrates networking capabilities

define → scan_port → (host, port)
  # In a full implementation, this would use tcp_connect
  store → result → tcp_connect(host, port)
  return → result
end

define → scan_range → (host, start, end)
  store → results → []
  
  for → port in range(start, end)
    store → status → scan_port(host, port)
    store → results → array_append(results, {"port": port, "status": status})
  end
  
  return → results
end

# Example usage
store → target → "192.168.1.1"
store → ports → scan_range(target, 1, 100)
print → ports
